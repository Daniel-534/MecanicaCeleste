\section{Hamiltoniano del Sistema}

\begin{equation}
\mathcal{H} = \frac{1}{2}\frac{p_x^2}{m} + \frac{1}{2}kx^2
\end{equation}

\subsection{Ecuación para el momento lineal}

\begin{equation}
\frac{dp_x}{dt} = -\frac{\partial\mathcal{H}}{\partial x} = -kx
\end{equation}

\subsection{Ecuación para la posición}

\begin{equation}
\frac{dx}{dt} = \frac{\partial\mathcal{H}}{\partial p_x} = \frac{p_x}{m}
\end{equation}
